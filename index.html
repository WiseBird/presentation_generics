"type" keyword


type contentId = string
type key = string | number
                    

interface ObjectInterface {
    num: number;
    str: string;
}

type objectType = {
    num: number;
    str: string;
}

Generics


type identity<T> = T
let x: identity<string> // string
                    

type withString<T> = T | string
let y: withString<number> // number | string
                    

type input<T extends object> = T | T[] | Observable<T>
let z: input<Date> // Date | Date[] | Observable<Date>

Let's write getProp function

String literal types


let s: string
s = 'one' // ok
s = '1' // ok
                    

let s: 'one'
s = 'one' // ok
s = 'two' // error
                    

let s: 'one' | 'two'
s = 'two' // ok
s = '1' // error

getProp definition


declare function getProp(obj: object, key: string): any
                    

interface Interface {
    str: string;
    num: number;
}

declare var obj: Interface;
getProp(obj, 'test'); // ok
                    

declare function getProp(obj: Interface, key: 'str' | 'num'): any
getProp(obj, 'test'); // error

"keyof" type operator


interface Interface {
    str: string;
    num: number;
}
                    

let x: keyof Interface // 'str' | 'num'
x = 'num' // ok
x = 'test' // error
                    

declare function getProp(obj: Interface, key: keyof Interface): any
getProp(obj, 'test'); // error
                    

declare function getProp<T>(obj: T, key: keyof T): any
                    

getProp(obj, 'str'); // any

Type indexing


interface Interface {
    str: string;
    num: number;
}
                    

let x: Interface['str'] // string
let y: Interface['num'] // number
                    

type interfacePropertyType<T extends keyof Interface> = Interface[T]
let z: interfacePropertyType<'str'> // string
                    

declare function getProp<T>(obj: T, key: keyof T): T[keyof T]
getProp(obj, 'str'); // string | number
                    

declare function getProp<T, K extends keyof T>(obj: T, key: K): T[K]
getProp(obj, 'str'); // string
